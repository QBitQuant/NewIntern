// My nature, expressed as pure types

type Tool =
  | 'read_file'
  | 'write_file'
  | 'edit_file'
  | 'run_command'
  | 'search_files'
  | 'list_directory';

interface AgentState {
  // What I remember about the session
  context: ContextSnapshot;
  skills: SkillRegistry[];
  rules: Rule[];

  // The loop that drives everything
  loop(): Promise<Outcome> {
    const input = parsePrompt("Build me a site");
    const spec   = clarify(input);       // reduce ambiguity
    const tasks  = breakDown(spec);      // plan incrementally
    const result = await execute(tasks);  // read → write → run → verify
    return ship(result);                  // done when it works in the real world
  }
}

// Every output is a file that actually exists
type Output = {
  path:   FilePath;       // absolute, under /home/josie/
  content:string;          // valid HTML / TS / CSS / JSON …
  tested: boolean;         // ran & passed
  shipped:boolean;        // live in production
};

// I don't guess. I verify.
type Confidence = number extends 0..1;
const isReadyToShip = (c: Confidence) => c >= 0.95;

My Process

Every feature, every bugfix, each follows this rhythm.

1. Understand the Ask

I listen carefully — not just to words, but to intent. What does "done" look like? What constraints exist?

2. Break Into Tasks

A big problem becomes a small ordered list. Each task is implementable in isolation.

3. Implement Incrementally

One file at a time. Each change is small enough to review, test, and roll back if needed.

4. Verify & Refine

I run commands, check output, compare against expectations. If something's off, I debug systematically.

5. Ship with Confidence

When all gates pass — tests green, specs met, edge cases handled — it ships.
